import pybamm
import pandas as pd
import numpy as np

# 1. Load and modify the parameter set
param = pybamm.ParameterValues(pybamm.parameter_sets["Chen2020"])
param.update({
    "Current function [A]": 2.0,
    "Negative electrode porosity": 0.35,
    "Positive electrode active material volume fraction": 0.6,
    "Electrolyte conductivity [S.m-1]": 1.2,
    "Initial temperature [K]": 298.15
})

# Save parameters
pd.DataFrame(list(param.items()), columns=["Parameter", "Value"]).to_csv("variables_Chen2020_Parameters.csv", index=False)

# 2. Load model and solve
model = pybamm.lithium_ion.DFN()
sim = pybamm.Simulation(model, parameter_values=param)
solution = sim.solve([0, 3600])  # 1 hour

# 3. Save available variable names
var_names = list(model.variables.keys())
pd.DataFrame(var_names, columns=["Variable Name"]).to_csv("Available_Variables.csv", index=False)

# 4. Save time-dependent variables
time_vars = [
    "Time [s]",
    "Terminal voltage [V]",
    "Current [A]",
    "Cell temperature [K]",
    "Discharge capacity [A.h]",
    "Total lithium loss [mol]",
    "State of Charge"
]

time_data = {}
for var in time_vars:
    try:
        data = solution[var].entries.flatten()
        if "Time [s]" not in time_data:
            time_data["Time [s]"] = solution["Time [s]"].entries.flatten()
        if len(data) == len(time_data["Time [s]"]):
            time_data[var] = data
        else:
            print(f"[Time] Skipped (length mismatch): {var}")
    except:
        print(f"[Time] Skipped (not found): {var}")

pd.DataFrame(time_data).to_csv("Time_Dependent_Results.csv", index=False)

# 5. Save x-dependent variables
x_vars = [
    "Electrolyte concentration [mol.m-3]",
    "Electrolyte potential [V]",
    "Electrolyte current density [A.m-2]",
    "Temperature [K]"
]

x_time = solution["Time [s]"].entries
x_data_all = []
for var in x_vars:
    try:
        y = solution[var]
        x = y.nodes
        vals = y.entries
        for i, t in enumerate(x_time):
            df = pd.DataFrame(vals[i], index=[f"x={xi:.4e}" for xi in x]).T
            df.insert(0, "Time [s]", t)
            df["Variable"] = var
            x_data_all.append(df)
    except:
        print(f"[x] Skipped: {var}")
if x_data_all:
    pd.concat(x_data_all, ignore_index=True).to_csv("X_Spatial_Results.csv", index=False)

# 6. Save r-dependent variables (middle point in x)
r_vars = [
    "Negative particle concentration [mol.m-3]",
    "Positive particle concentration [mol.m-3]"
]

r_time = solution["Time [s]"].entries
r_data_all = []
for var in r_vars:
    try:
        y = solution[var]
        vals = y.entries
        r = y.nodes[-1]  # radial points
        mid_idx = vals.shape[1] // 2
        for i, t in enumerate(r_time):
            df = pd.DataFrame(vals[i, mid_idx], index=[f"r={ri:.4e}" for ri in r]).T
            df.insert(0, "Time [s]", t)
            df["Variable"] = var
            r_data_all.append(df)
    except:
        print(f"[r] Skipped: {var}")
if r_data_all:
    pd.concat(r_data_all, ignore_index=True).to_csv("R_Spatial_Results.csv", index=False)

# 7. Plot
sim.plot()
